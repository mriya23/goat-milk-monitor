function assessQuality(pH, mq135, rgb = "245,250,252") {
    // Convert RGB string to colour average
    const rgbValues = rgb.split(",").map(v => parseInt(v.trim()));
    const colour = Math.round((rgbValues[0] + rgbValues[1] + rgbValues[2]) / 3);
